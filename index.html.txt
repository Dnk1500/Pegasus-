  <script>
    const MONTHLY_RATE = 0.044; // 4,4% mensual (fijo)

    const eur = new Intl.NumberFormat("es-ES", {
      style: "currency",
      currency: "EUR",
      maximumFractionDigits: 2
    });

    function toNumber(value) {
      const n = Number(value);
      return Number.isFinite(n) ? n : 0;
    }

    function calculate() {
      const investment = toNumber(document.getElementById("investment").value);

      const monthly = investment * MONTHLY_RATE;
      const yearly  = monthly * 12;

      const endMonth = investment + monthly;
      const endYear  = investment + yearly;

      document.getElementById("monthly").innerText  = eur.format(monthly);
      document.getElementById("yearly").innerText   = eur.format(yearly);
      document.getElementById("endMonth").innerText = eur.format(endMonth);
      document.getElementById("endYear").innerText  = eur.format(endYear);
    }
